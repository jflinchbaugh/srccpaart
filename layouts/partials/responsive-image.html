{{- $path := .path -}} {{- $img := resources.Get (printf "static/%s" $path) -}} {{- if $img -}} {{- $targetWidth := 1000 -}} {{- if lt $img.Width $targetWidth -}} {{- $targetWidth = $img.Width -}} {{- end -}} {{- $fallback := $img -}} {{- if lt $targetWidth $img.Width -}} {{- $fallback = $img.Resize (printf "%dx q80" $targetWidth) -}} {{- end -}} {{- $webp := $img.Resize (printf "%dx webp q75" $targetWidth) -}} {{- $srcset := printf "%s 1x" $webp.RelPermalink -}} {{- $targetWidth2x := mul $targetWidth 2 -}} {{- if lt $targetWidth2x $img.Width -}} {{- $webp2x := $img.Resize (printf "%dx webp q75" $targetWidth2x) -}} {{- $srcset = printf "%s, %s 2x" $srcset $webp2x.RelPermalink -}} {{- else if lt $targetWidth $img.Width -}} {{- $webp2x := $img.Resize (printf "%dx webp q75" $img.Width) -}} {{- $srcset = printf "%s, %s 2x" $srcset $webp2x.RelPermalink -}} {{- end -}} {{- else -}} {{- warnf "missing image: static/%s (referenced with alt %q)" $path .alt -}} {{- end -}}
